# Explore dataset structure
def explore_dataset(base_path):
    """Explore the dataset structure and extract class information"""

    video_extensions = ['.mp4', '.avi', '.mov', '.mkv', '.wmv']
    class_info = {}

    for root, dirs, files in os.walk(base_path):
        for file in files:
            if any(file.lower().endswith(ext) for ext in video_extensions):
                class_name = os.path.basename(root)
                if class_name not in class_info:
                    class_info[class_name] = []
                class_info[class_name].append(os.path.join(root, file))

    return class_info

# Cell 7: Create dataset from videos
def create_video_dataset(dataset_info, samples_per_class=20, sequence_length=15):
    """Create dataset from video files"""
    video_processor = VideoProcessor(sequence_length=sequence_length)

    X = []
    y = []
    class_mapping = {}
    reverse_mapping = {}

    # Create class mapping
    for idx, class_name in enumerate(dataset_info.keys()):
        class_mapping[class_name] = idx
        reverse_mapping[idx] = class_name

    print("🔄 Processing videos and extracting features...")

    for class_name, video_paths in dataset_info.items():
        print(f"Processing class: {class_name}")
        class_id = class_mapping[class_name]

        # Limit samples per class
        video_samples = video_paths[:samples_per_class]

        for video_path in video_samples:
            try:
                sequence = video_processor.extract_features_from_video(video_path)
                X.append(sequence)
                y.append(class_id)
                print(f"  ✅ Processed: {os.path.basename(video_path)}")
            except Exception as e:
                print(f"  ❌ Error processing {video_path}: {e}")
                continue

    video_processor.close()

    X = np.array(X)
    y = np.array(y)

    print(f"✅ Dataset created: {X.shape} sequences, {len(np.unique(y))} classes")
    return X, y, class_mapping, reverse_mapping
